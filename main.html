async function sendAction(type) { const name = localStorage.getItem("user_name"); const role = localStorage.getItem("user_role"); const island = localStorage.getItem("user_island"); if (!confirm(name + " 해설사님, [" + type + "] 기록하시겠습니까?")) return; try { const res = await fetch('https://bdgeopark.onrender.com/record', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ name: name, island: island, role: role, action: type // "출근" 또는 "퇴근"이 '특이사항' 컬럼으로 들어갑니다. }) }); if (res.ok) { alert(type + " 기록 완료!"); } else { alert("기록 실패!"); } } catch (error) { alert("네트워크 오류!"); } }
